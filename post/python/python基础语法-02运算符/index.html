#!/usr/bin/env python
# -*- encoding: utf-8 -*-

'''
    python 比较运算符
'''

a = 21
b = 10
c = 0

if (a == b):
    print "1 - a 等于 b"
else:
    print "1 - a 不等于 b"

if ( a != b):
    print "2 - a 不等于 b"
else:
    print "2 - a 等于 b"

if (a <> b):
    print "3 - a 不等于 b"
else:
    print "3 - a 等于 b"

if (a < b):
    print "4 - a 小于 b"
else:
    print "4 - a 大于等于 b"

if (a > b):
    print "5 - a 大于 b"
else:
    print "5 - a 小于等于 b"

    修改变量a 和 b 的值
